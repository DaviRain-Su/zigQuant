        var best: ?struct { exchange: IExchange, price: Decimal } = null;
        
        for (self.registry.getConnectedExchanges()) |exchange| {
            const ticker = exchange.getTicker(pair) catch continue;
            
            const price = if (side == .buy) ticker.ask else ticker.bid;

            const fill_amount = Decimal.min(remaining, level.quantity);

                .side = request.side,
                .order_type = request.order_type,
                .amount = fill_amount,
                .price = level.price,
            });
            
            try orders.append(.{
                .order = order,
                .exchange = level.exchange,
            });
        }
        
        return .{
            .orders = orders.toOwnedSlice() catch &.{},
            .routing_strategy = .split,
            .unfilled_amount = remaining,
        };
